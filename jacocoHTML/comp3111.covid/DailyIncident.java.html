package comp3111.covid;

/**
 * This class stores data of one location in one day.
 * Including date, total cases(per million), total deaths(per million), and people fully vaccinated(per hundred).
 */
public class DailyIncident {
	private String date;
	private String total_cases;
	private String total_cases_per_million;
	private String total_deaths;
	private String total_deaths_per_million;
	private String people_fully_vaccinated;
	private String people_vaccinated_per_hundred;
	
	/**
	 * Initializer
	 * @param date date of recording
	 * @param total_cases total cases on that date
	 * @param total_cases_per_million total deaths per million on that date
	 * @param total_deaths total deaths on that  date
	 * @param total_deaths_per_million total deaths per million on that date
	 * @param people_fully_vaccinated people fullt vaccinated on that date
	 * @param people_vaccinated_per_hundred people vaccinated per hundred on that date.
	 */
	public DailyIncident(String date,
			String total_cases, String total_cases_per_million,
			String total_deaths,String total_deaths_per_million,
			String people_fully_vaccinated,String people_vaccinated_per_hundred)
	{
		this.date = date;
		this.total_cases = total_cases;
		this.total_cases_per_million = total_cases_per_million;
		this.total_deaths = total_deaths;
		this.total_deaths_per_million = total_deaths_per_million;
		this.people_fully_vaccinated = people_fully_vaccinated;
		this.people_vaccinated_per_hundred = people_vaccinated_per_hundred;
	}
	
	/**
	 * @return return date in {@code String}, with format mm/dd/yyyy(e.g. 2/1/2020).
	 */
	public String getDate()
	{
		return this.date;
	}
	
	/**
	 * @return return total cases in {@code String}.
	 */
	public String getTotalCases()
	{
		return this.total_cases;
	}
	
	/**
	 * @return return total cases per million in {@code String}.
	 */
	public String getTotalCasesPerMillion()
	{
		return this.total_cases_per_million;
	}
	
	/**
	 * @return return total deaths in {@code String}.
	 */
	public String getTotalDeaths()
	{
		return this.total_deaths;
	}
	
	/**
	 * @return return total deaths per million in {@code String}.
	 */
	public String getTotalDeathsPerMillion()
	{
		return this.total_deaths_per_million;
	}
	
	/**
	 * @return return people fully vaccinated in {@code String}.
	 */
	public String getPeopleFullyVaccinated()
	{
		return this.people_fully_vaccinated;
	}
	
	/**
	 * @return return people vaccinated per hundred in {@code String}.
	 */
	public String getPeopleVaccinatedPerHundred()
	{
		return this.people_vaccinated_per_hundred;
	}
	
	
	/**
	 * Add all the parameters together as {@code String} used to make comparison between {@code DailyIncident} objects in unit test.
	 * @return
	 */
	@Override
	public String toString() {
		return this.date+this.people_fully_vaccinated+this.people_vaccinated_per_hundred+this.total_cases+this.total_cases_per_million+this.total_deaths+this.total_deaths_per_million;
	}
	
	

}
