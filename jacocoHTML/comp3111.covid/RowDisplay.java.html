package comp3111.covid;

import javafx.beans.property.SimpleStringProperty;

/**
 * this class is used to store a row of the result table
 * one row have three columns, therefore {@code RowDisplay} is formed as three {@code String}s for the three columns
 * @author ZHANG,Hao
 */
public class RowDisplay {
 private final SimpleStringProperty firstCol;
    private final SimpleStringProperty secondCol;
    private final SimpleStringProperty thirdCol;

    public RowDisplay(String firstCol, String secondCol, String thirdCol) {
        this.firstCol = new SimpleStringProperty(firstCol);
        this.secondCol = new SimpleStringProperty(secondCol);
        this.thirdCol = new SimpleStringProperty(thirdCol);
    }

    public String getFirstCol() {
        return firstCol.get();
    }

    public void setFirstCol(String str) {
     firstCol.set(str);
    }

    public String getSecondCol() {
        return secondCol.get();
    }

    public void setSecondCol(String str) {
     secondCol.set(str);
    }

    public String getThirdCol() {
        return thirdCol.get();
    }

    public void TotalCasesPerM(String str) {
     thirdCol.set(str);
    }
}
