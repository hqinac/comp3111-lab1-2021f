/**
 * A connecttion between the total UI to the UI of different Tasks
 * @author qinhanmo
 *
 */
public class StartQueryTable{
	public String selectedDataset;
	private boolean isCheckPass;
	private String fxmlFile="";
	
	/**
	 * initializer
	 * @param fileName name of the input csv file
	 */
	public StartQueryTable(String fileName){
		selectedDataset = "";
		isCheckPass  = false;
		fxmlFile = fileName;
	}
	
	/**
	 * define a {@code CSVParser} to achieve the csv file
	 * @param dataset name of the csv file in string
	 * @return {@code CSVParser}
	 */
	public static CSVParser getFileParser(String dataset) {
	     FileResource fr = new FileResource("dataset/" + dataset);
	     return fr.getCSVParser(true);
	}
	
	/**
	 * Check whether the input csv file exists
	 * @param dataset name of the csv file in {@code String}.
	 * @return true if exists, false if not exists.
	 */
	public static boolean isDatasetValid(String dataset)
	{
		File Set = new File("./src/main/resources/dataset/" + dataset);
		return Set.exists();
	}
	
	/**
	 * Store the input dataset name(name of the csv file)
	 * @param DataSet the input dataset name(name of the csv file)
	 */
	public void storeDataSet(String DataSet) {
		selectedDataset = DataSet;
	}
	
	/**
	 * Store the result of {@link #isDatasetValid(String)}, to extrict the change of input dataset after correct checking.
	 */
	public void checkPass() { isCheckPass = true;}
	
	
	/**
	 * function to open the UI for task parameters(date, country) initialization.
	 */
	public void startANewWindow() {
	    try {
	        FXMLLoader loader = new FXMLLoader();
	        loader.setLocation(getClass().getResource("/"+fxmlFile));
	        TitledPane pane = loader.load();

	        Scene scene = new Scene(pane);
	        Stage stage = new Stage();
	        stage.setScene(scene);
	        stage.show();

	    } catch (Exception e) {
	    	System.out.println("Unable to init layout : " + e.getMessage());
	    }
	}
}
