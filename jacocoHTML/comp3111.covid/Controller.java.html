    /**
     * This method means click the "click to start"tab of TableB to open the GUI of illitialization.(popUpTableB.fxml)
     * Check the input selectedDataset is available before opening, if the data set is invalid, give an Alert of error.
     * @param event Operation on GUI
     */
	@FXML
	void clickToStartTableB(ActionEvent event) {
		String selectedDataset = tf_dataset_Tb.getText();
		StartQueryTable startTableB = new StartQueryTable("popUpTableB.fxml");


		if(!startTableB.isDatasetValid(selectedDataset)) {
			Alert a = new Alert(AlertType.NONE); 
			a.setAlertType(AlertType.ERROR); 
			a.setContentText("Sorry, the input DataSet doesn't exist.");  
			a.show();
			return;
		}
		else {   
			startTableB.storeDataSet(selectedDataset);  
			startTableB.checkPass();
			startTableB.startANewWindow();
			GlobalDataSet.dataSetTableB = selectedDataset;
		}
	}
